		name = decolonization.2.a

        }
		if = {
	        limit = {
		        country_exists = NIA

	        limit = {
		        country_exists = CMR
			    NOT = { country_exists = NIA }
	        }
	        CMR = { transfer_state = 1105 transfer_state = 1106 }
        }
	}
}

#The Malayan Question
country_event = {
	id = decolonization.10
	title = decolonization.10.t	
	desc = decolonization.10.d
	
	is_triggered_only = yes
	fire_only_once = yes
	
	#Malaysia Plan
	option = {
		name = decolonization.10.a
		ai_chance = {
			factor = 99
		}
		MAL = {
			drop_cosmetic_tag = yes
			transfer_state = 336
			transfer_state = 333
			transfer_state = 785
		    if = {
		        limit = {
			        711 = { is_core_of = SGP }
		        }
		        MAL = { transfer_state = 711 }
				711 = { add_core_of = MAL }
	        }
			if = {
		        limit = {
			        712 = { is_core_of = SGP }
		        }
		        MAL = { transfer_state = 712 }
				712 = { add_core_of = MAL }
	        }
		}
	}
	
    #North Borneo Plan
	option = {
		name = decolonization.10.b
		ai_chance = {
			factor = 1
		}
		set_autonomy = { target = BRU autonomy_state = autonomy_free }	
		MAL = {
			drop_cosmetic_tag = yes
			transfer_state = 336
		    if = {
		        limit = {
			        711 = { is_core_of = SGP }
		        }
		        MAL = { transfer_state = 711 }
				711 = { add_core_of = MAL }
	        }
			if = {
		        limit = {
			        712 = { is_core_of = SGP }
		        }
		        MAL = { transfer_state = 712 }
				712 = { add_core_of = MAL }
	        }
		}
		BRU = {
		    set_politics = {
				ruling_party = socialist
				elections_allowed = yes
			}
			add_popularity ={
		        ideology = socialist
		        popularity = 0.9
		    }
			set_cosmetic_tag = NBR
			transfer_state = 333
			transfer_state = 785
			333 = { add_core_of = BRU remove_core_of = MAL }
			785 = { add_core_of = BRU remove_core_of = MAL }
		}
	}
}
