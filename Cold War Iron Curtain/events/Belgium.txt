	}
}

 # Declare War event
country_event = {
	id = belgium.4
	title = belgium.4
	desc = belgium.4
	picture = GFX_BEL_Civil_War
	
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			BEL = {
				delete_unit_template_and_units = {
					division_template = "Division d'Infanterie"
					division_template = "Division de Cavalerie"
					division_template = "Cavalerie Motorisée"
					division_template = "Division Coloniale"
				}
			}
			FLA = { load_oob = FLA_1949 }
			WAL = { load_oob = WAL_1949 }
		}
	}
	
	option = {
		name = belgium.4.a
		if = {
			limit = {
				BEL = { has_country_flag = BEL_Flemish_Civil_War }
			}
			FLA = {
				declare_war_on = {
					target = WAL
					type = civil_war
				}
			}
			break = yes
		}
		if = {
			limit = {
				BEL = { has_country_flag = BEL_Walloon_Civil_War }
			}
			WAL = {
				declare_war_on = {
					target = FLA
					type = civil_war
				}
			}
			break = yes
		}
	}
}
