#Surrender when Israel has taken too much land in Jordania
country_event = {
	id = jordania.1
	title = jordania.1.t
	desc = jordania.1.d
	picture = GFX_report_event_soviet_tanks_snow
	

	is_triggered_only = yes

	#Surrender and offer territory
	option = {

		}

			ISR = { 
				white_peace = {
					tag = JOR

			}
			transfer_state = 781
		}
	}
}
