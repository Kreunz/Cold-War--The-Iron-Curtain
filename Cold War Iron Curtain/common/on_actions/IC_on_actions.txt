on_actions = {
	on_annex = {
		effect = {
			if  = {
				limit = {
					any_owned_state = {
						is_core_of = FROM
						NOT = { is_core_of = ROOT }
						NOT = { has_state_flag = active_insurgency }					
					}
				}
				every_owned_state = {
					limit = {
						is_core_of = FROM
						NOT = { is_core_of = ROOT }
						NOT = { has_state_flag = active_insurgency }
					}
					set_state_flag = active_insurgency
					set_state_flag = anti_colonial_insurgency
					set_variable = { insurgency_strength = 20 }
					add_state_modifier = {
						modifier = {
							local_resources = -0.20
							local_supplies = -0.30
							local_intel_to_enemies = 0.25
							local_factory_sabotage = 0.10
							local_non_core_manpower = -0.30
							attrition = 0.30
							local_org_regain = -0.25
						}
					}
				}
			}
		}
	}


	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if  = {
				limit = {
					any_owned_state = {
						is_core_of = FROM
						NOT = { is_core_of = ROOT }
						NOT = { has_state_flag = active_insurgency }					
					}
				}
				every_owned_state = {
					limit = {
						is_core_of = FROM
						NOT = { is_core_of = ROOT }
						NOT = { has_state_flag = active_insurgency }
					}
					set_state_flag = active_insurgency
					set_state_flag = anti_colonial_insurgency
					set_variable = { insurgency_strength = 20 }
					add_state_modifier = {
						modifier = {
							local_resources = -0.20
							local_supplies = -0.30
							local_intel_to_enemies = 0.25
							local_factory_sabotage = 0.10
							local_non_core_manpower = -0.30
							attrition = 0.30
							local_org_regain = -0.25
						}
					}
				}
			}
		}
	}
	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					has_war_with = FROM
				}
				damage_building = {
					type = infrastructure
					damage = 5
				}
				damage_building = {
					type = industrial_complex
					damage = 5
				}
				damage_building = {
					type = arms_factory
					damage = 5
				}
				remove_pop_scale = yes
			}
		}
		if = {
			limit = {
				OR = {
					FROM = { original_tag = ROOT }
					AND = {
						original_tag = VIN
						FROM = { original_tag = VIE }
					}
					AND = {
						original_tag = VIE
						FROM = { original_tag = VIN }
					}
					AND = {
						original_tag = CHI
						FROM = { original_tag = PRC }
					}
					AND = {
						original_tag = PRC
						FROM = { original_tag = CHI }
					}
					AND = {
						original_tag = KPA
						FROM = { original_tag = KOR }
					}
					AND = {
						original_tag = KOR
						FROM = { original_tag = KPA }
					}
					AND = {
						original_tag = CAM
						FROM = { original_tag = CMB }
					}
					AND = {
						original_tag = CMB
						FROM = { original_tag = CAM }
					}
					AND = {
						original_tag = DDR
						FROM = { original_tag = WGR }
					}
					AND = {
						original_tag = WGR
						FROM = { original_tag = DDR }
					}
					AND = {
						original_tag = GRE
						FROM = { original_tag = PDG }
					}
					AND = {
						original_tag = PDG
						FROM = { original_tag = GRE }
					}
				}
			}
			country_event = ic_civil_war.1
		}
	}
	on_nuke_drop = {

		effect = {
			every_country = {
				limit = { 
					NOT = { has_idea = disarmed_nation_nuke }
				}
				add_ideas = armed_nation_nuke
			}
		}
	}
}
