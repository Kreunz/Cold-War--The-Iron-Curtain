	egypt_view = {
		context_type = player_context
		
		window_name = "egypt_view_container"
		visible = {
			 check_variable = { egy_show > 0.0 }
		}	
		effects = {
			sanction_egy_click = {
                add_opinion_modifier = { target = EGY modifier = totalembargo }
                add_opinion_modifier = { target = EGY modifier = trade_sanctions }	
				add_political_power = -20
				set_variable = { egy_sanction = 1 }
			}
			embargo_egy_click = {
                add_opinion_modifier = { target = EGY modifier = trade_sanctions }	
				add_political_power = -20
				set_variable = { egy_sanction = 1 }
			}
			normalize_trade_relations_egy_click = {
				
				remove_opinion_modifier = { target = EGY modifier = totalembargo }
				remove_opinion_modifier = { target = EGY modifier = trade_sanctions }					
				add_political_power = -20
				set_variable = { egy_sanction = 0 }
			}
			invest_in_egy_economy_click = {
				EGY = {
					country_event = { id = build_building.1 days = 50  }
				}
				subtract_from_variable = { var = money value = 500  }
				add_opinion_modifier = { target = EGY  modifier = sent_economic_aid }
			}
			send_egy_manpower_click = {
				EGY = {
					add_manpower = 10000
				}
				add_manpower = -10000
				add_opinion_modifier = { target = EGY  modifier = sent_manpower }
			}
			send_infantry_equipment_egy_click = {
				send_equipment = { target = EGY type = infantry_equipment amount = 500 }	
				add_opinion_modifier = { target = EGY  modifier = sent_equipment }
			}
			send_mbt_egy_click = {
				send_equipment = { target = EGY type = mbt_equipment amount = 500 }	
				add_opinion_modifier = { target = EGY  modifier = sent_equipment }
			}
			send_fighter_equipment_egy_click = {
				send_equipment = { target = EGY type = jet_multirole_equipment amount = 500 }	
				add_opinion_modifier = { target = EGY  modifier = sent_equipment }
			}
			send_money_egy_click = {
				EGY = { add_to_variable = { var = money value = 500  } }
				subtract_from_variable = { var = money value = 500  }
				add_opinion_modifier = { target = EGY  modifier = sent_money }
			}
			retake_our_cores_egy_click = {
				set_country_flag = cultural_unification_war_aggressor
				EGY = { set_country_flag = cultural_unification_war_defender }
				declare_war_on = { target = EGY type = take_core_state }	
			}
			liberation_war_egy_click = {
				set_country_flag = badboy_war_attacker
				EGY = { set_country_flag = badboy_war_defender }
				declare_war_on = { target = EGY type = liberate_wargoal }	
			}
			exitbutton_egy_click = {
				set_variable = { egy_show = 0.0 }
			}		
		}
